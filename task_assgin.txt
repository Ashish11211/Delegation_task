<script>
  var data;
    function onFailure(error) {
        console.log(error.message)
      }

    function removeActiveClasses() {
      document.getElementById("follow-up-link").classList.remove("active");
      document.getElementById("add-new-task-link").classList.remove("active");
      document.getElementById("all-pending-tasks").classList.remove("active");
    }

    function loadFView(){
      loadingStart();
      removeActiveClasses();
      loadView({func : "loadFollowUpView", callback: setDataForSearch});
      document.getElementById("follow-up-link").classList.add("active");
      loadingEnd();
    }

    function loadAddView(){
      loadingStart();
      removeActiveClasses();
      loadView({func : "loadAddNewView", callback:setDoerNames});
      document.getElementById("add-new-task-link").classList.add("active");
      loadingEnd();
    }

    function loadAllPendingView() {
      loadingStart();
      removeActiveClasses();
      loadView({func : "loadPendingTasksView", callback:setSearchDataforAllPending});
      document.getElementById("all-pending-tasks").classList.add("active");
      loadingEnd();
    }


    function setDoerNames() {
        loadingStart();
        google.script.run.withSuccessHandler(function(names) {
                var dropdown = document.getElementById("name");
                
                // Clear existing options (except the first one)
                while (dropdown.options.length > 1) {
                    dropdown.remove(1);
                }

                // Add new options
                names.forEach(function(name) {
                    var option = document.createElement("option");
                    option.textContent = name;
                    option.value = name;
                    dropdown.appendChild(option);
                });
                
                loadingEnd();
            })
            .withFailureHandler(onFailure)
            .getAllDoers();
    }


    function otherFunc(params){
      document.getElementById("h1").textContent = params.title
    }

    function loadView(options){
      var id = typeof options.id === "undefined" ? "app" : options.id
      var cb = typeof options.callback === "undefined" ? function(){} : options.callback

      google.script.run.withSuccessHandler(function(html){
        document.getElementById(id).innerHTML = html
        typeof options.params === "undefined" ? cb() : cb(options.params)
      }).withFailureHandler(onFailure)[options.func]();

    }

    function setSearchDataforAllPending(){
      loadingStart()
      google.script.run.withSuccessHandler(function(dataReturned){
        var searchInput = document.getElementById("searchName").value.toString().toLowerCase() 
        var taskInput = document.getElementById("searchTask").value.toString().toLowerCase()

        data = dataReturned.slice()

        if (!searchInput && !taskInput) {
          var searchResultsBox = document.getElementById("searchResults")
          var templateBox = document.getElementById("rowTemplate")
          var template = templateBox.content


       searchResultsBox.innerHTML = ""

      if (data.length < 1) {
        document.querySelector('.no-results').classList.remove("invisible")
      } else {
        document.querySelector('.no-results').classList.add("invisible")
      }

      data.forEach(function(r){
        var tr = template.cloneNode(true)
        var taskIdCol = tr.querySelector(".taskID")
        var nameCol = tr.querySelector(".doerName")
        var taskCol = tr.querySelector(".task")
        var plDate = tr.querySelector(".planned_date")
        var revise = tr.querySelector(".revise-button")
        var done = tr.querySelector(".confirm-button2")
        done.dataset.taskid = r[0]
        revise.dataset.taskid = r[0]
        taskIdCol.textContent = r[0]
        nameCol.textContent = r[1]
        taskCol.textContent = r[2]
        plDate.textContent = r[3]
        searchResultsBox.appendChild(tr)
      })
        }
      loadingEnd()
      }).withFailureHandler(onFailure).fetchAllPendingData()
      
    }

    function setDataForSearch(){
      loadingStart()
      google.script.run.withSuccessHandler(function(dataReturned){
        var searchInput = document.getElementById("searchName").value.toString().toLowerCase() 
        var taskInput = document.getElementById("searchTask").value.toString().toLowerCase()

        data = dataReturned.slice()

        if (!searchInput && !taskInput) {
          var searchResultsBox = document.getElementById("searchResults")

       searchResultsBox.innerHTML = ""

      if (data.length < 1) {
        document.querySelector('.no-results').classList.remove("invisible")
      } else {

        searchResultsBox.appendChild(tr)
      })
        }
      loadingEnd()
      }).withFailureHandler(onFailure).fetchRelevantData()
      
    }

    function search(){
      var searchInput = document.getElementById("searchName").value.toString().toLowerCase() 
      var taskInput = document.getElementById("searchTask").value.toString().toLowerCase()
      searchInput = searchInput === "" ? "undefined" : searchInput
      taskInput = taskInput === "" ? "undefined" : taskInput


      var resultsArray = data.filter(function(r){
        var dataName = r[1].toString().toLowerCase()
        var dataTask = r[2].toString().toLowerCase()

        return dataName.indexOf(searchInput) !== -1 ||dataTask.indexOf(taskInput) !== -1
      })

      var searchResultsBox = document.getElementById("searchResults")
      var templateBox = document.getElementById("rowTemplate")
      var template = templateBox.content


      searchResultsBox.innerHTML = ""

      resultsArray.forEach(function(r){
        var tr = template.cloneNode(true)
        var taskIdCol = tr.querySelector(".taskID")
        var nameCol = tr.querySelector(".doerName")
        var taskCol = tr.querySelector(".task")
        var plDate = tr.querySelector(".planned_date")
        var revise = tr.querySelector(".revise-button")
        var done = tr.querySelector(".confirm-button")
        if (!done) {
          done = tr.querySelector(".confirm-button2");
        }
        done.dataset.taskid = r[0]
        revise.dataset.taskid = r[0]
        taskIdCol.textContent = r[0]
        nameCol.textContent = r[1]
        taskCol.textContent = r[2]
        plDate.textContent = r[3]
        searchResultsBox.appendChild(tr)
      })

      if (searchInput === "undefined" && taskInput === "undefined") {
        data.forEach(function(r){
        var tr = template.cloneNode(true)
        var taskIdCol = tr.querySelector(".taskID")
        var nameCol = tr.querySelector(".doerName")
        var taskCol = tr.querySelector(".task")
        var plDate = tr.querySelector(".planned_date")
        var revise = tr.querySelector(".revise-button")
        var done = tr.querySelector(".confirm-button")
        done.dataset.taskid = r[0]
        revise.dataset.taskid = r[0]
        taskIdCol.textContent = r[0]
        nameCol.textContent = r[1]
        taskCol.textContent = r[2]
        plDate.textContent = r[3]
        searchResultsBox.appendChild(tr)
      })
      }


    }

    function reviseTheTask(returnTo) {
      loadingStart("1");
      var taskDetails = {};
      taskDetails.id = document.getElementById("id").value;
      taskDetails.pl = document.getElementById("planned_date").value;

      if (!taskDetails.pl || !taskDetails.id) {
        loadingEnd();
        alert("Please enter all the details");
      } else {
      google.script.run.withSuccessHandler(function(){
        if (returnTo === "loadAllPendingView") {
        loadAllPendingView();
        } else {
          loadFView();
        }
      }).withFailureHandler(onFailure).addNext(taskDetails);
      }
    }

    function addTask(){
      var taskInfo = {}
      taskInfo.name = document.getElementById("name").value
      taskInfo.task = document.getElementById("task").value
      taskInfo.fdate = document.getElementById("planned_date").value

      if (!taskInfo.name || !taskInfo.task || !taskInfo.fdate){
        alert("Please fill all the details.")
      } else {
        loadingStart()
        google.script.run.withSuccessHandler(function(){
          document.getElementById("name").value = ""
          document.getElementById("task").value = ""
          document.getElementById("planned_date").value = ""
          document.getElementById("success-message").classList.remove("invisible")
          loadingEnd()
          setTimeout(function(){
            document.getElementById("success-message").classList.add("invisible")
          },3000)

        }).withFailureHandler(onFailure).addNew(taskInfo)

      }
    }

    function completeTask(id, returnTo) {
      loadingStart();
      google.script.run.withSuccessHandler(function(r){
      if (returnTo === "loadAllPendingView") {
        loadAllPendingView();
      } else {
        loadFView();
      }
      loadingEnd();
      }).withFailureHandler(onFailure).taskComplete(id);
    }

    function displayConfirmation(e){
      if(e.target.dataset.buttonState === "done"){
      e.target.nextElementSibling.classList.remove("d-none")
      e.target.dataset.buttonState = "cancel"
      e.target.textContent = "Cancel"
      } else {
      e.target.nextElementSibling.classList.add("d-none")
      e.target.dataset.buttonState = "done"
      e.target.textContent = "Done"
      }
    }

    function loadingStart(t){
      t ? console.log(t) : t
      document.getElementById("loading").classList.remove("invisible")
    }

    function loadingEnd(){
      document.getElementById("loading").classList.add("invisible")
    }

    function afterReviseLoads(params){
      loadingStart()
      google.script.run.withSuccessHandler(function(r){
        document.getElementById("name").value = r.name
        document.getElementById("task").value = r.task
        document.getElementById("id").value = r.taskid
        loadingEnd()
      }).withFailureHandler(onFailure).getInfoByTaskID(params.taskID)
    }

    document.getElementById("follow-up-link").addEventListener("click",loadFView)
    document.getElementById("add-new-task-link").addEventListener("click",loadAddView)
    document.getElementById("all-pending-tasks").addEventListener("click",loadAllPendingView)

    
  function clickEventHandler(e){
    if (e.target.matches("#add-task-button")){
     document.getElementById("add-task-button").disabled = "true";
      addTask();
      document.getElementById("add-task-button").disabled = "false";
    }
    if (e.target.matches(".revise-button")){
      loadView({func : "loadReviseView", callback:afterReviseLoads, params : {taskID:e.target.dataset.taskid}});
   }
    if (e.target.matches("#revise-task-button")){
      document.getElementById("revise-task-button").disabled = "true";
      reviseTheTask("loadAllPendingView"); // Pass the correct return view
    }
    if (e.target.matches(".done-button")){
     displayConfirmation(e);
    }
    if (e.target.matches(".confirm-button")){
      completeTask(e.target.dataset.taskid, "loadFView"); // Pass the correct return view
    }

    if (e.target.matches(".confirm-button2")){
      completeTask(e.target.dataset.taskid, "loadAllPendingView"); // Pass the correct return view
    }
  }


    function inputEventHandler(e){
      if (e.target.matches("#searchName") || e.target.matches("#searchTask")){
        search()
      }
    }
    document.getElementById("app").addEventListener("input",inputEventHandler)

    document.getElementById("app").addEventListener("click",clickEventHandler)

    document.addEventListener("DOMContentLoaded",loadFView)
